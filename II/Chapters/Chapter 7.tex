\chapter{Differential Geometry}


\section{Differential Forms}


We fix a finite dimensional vector space $E$ over $\RR$.
We denote by $E^*$ the dual vector space of $E$.

\begin{definitionenv}
    Let $U \subseteq E$ be an open subset, $p \in \NN$.
    We call $p$-form of $U$ any mapping $\omega$ from $U$ to $\bigwedge^p (E^*)$.
    For any $l \in \NN \cup\{\infty\}$, denote by $\Omega^p(U)$ the vector space of $p$-forms on $U$ which are of class $C^l$.

    \quad If $\omega: U \rightarrow \bigwedge^p(E^*)$ is a $p$-form, we define the support of $\omega$ the closure of
    $$\{x \in U \mid \omega(x) \neq 0\}$$
    in $U$, denote as $\supp(\omega)$.
    If $\supp(\omega)$ is compact, we say that $\omega$ has a compact support.
    Let $\Omega_{l,c}^{p}(U)$ be the set of $p$-forms of class $C^l$ on $U$ which are of compact support.

    \quad Let $\alpha,\beta$ be $p$-form and $q$-form respectively.
    We denote by
    $$\begin{array}{rrcl}
        \alpha \wedge \beta : & U & \longrightarrow & \bigwedge^{p+q}(E^*)\\
        & x & \longmapsto & \alpha(x) \wedge \beta(x).
    \end{array}$$
    One has $\beta \wedge \alpha = (-1)^{pq} \alpha \wedge \beta$.
\end{definitionenv}

\begin{remark}
    Let $E$ be a finite dimensional vector space over $\RR$,
    $p_1,\cdots,p_r$ be natural numbers,
    $$\begin{array}{rcl}
        \bigwedge^{p_1} (E^*) \times \cdots \times \bigwedge^{p_r} (E^*) & \longrightarrow & \bigwedge^{p_1+\cdots+p_r} (E^*)\\
        (\alpha_1,\cdots,\alpha_r) & \longmapsto & \alpha_1 \wedge \cdots \wedge \alpha_r
    \end{array}$$
    is multilinear, so it is of class $C^{\infty}$.

    Let $U \subseteq E$ be an open subset.
    For any $i \in \{1,\cdots,r\}$, let $\omega_i$ be a $p$-form on $U$.
    Let $\omega = \omega_1 \wedge \cdots \wedge \omega_r$.
    If $\omega_1,\cdots,\omega_r$ are of class $C^l$, so is $\omega$.
\end{remark}

\begin{propositionenv}
    Let $K$ be a commutative ring, $V$ and $W$ be $K$-modules.
    One has a homomorphism of $K$-modules,
    $$\begin{array}{rrcl}
        \Phi: & V^\vee \otimes W & \longrightarrow & \Hom_K (V,W)\\
        & \varphi \otimes w & \longmapsto & (x \mapsto \varphi(x) w).
    \end{array}$$
    It is an isomorphism if $V$ is a free $K$-module of finite rank.
\end{propositionenv}
\begin{proofenv}
    The mapping
    $$\begin{array}{rcl}
        V^\vee \times W & \longrightarrow & \Hom_K (V,W)\\
        (\varphi,w) & \longmapsto & (x \mapsto \varphi(x) w)
    \end{array}$$
    is $K$-bilinear, hence it induced a homomorphism of $K$-modules.
    Suppose that $(e_i)_{i=1}^n$ is a basis of $V$, let $(e_i^\vee)_{i=1}^n$ be the dual basis of $(e_i)_{i=1}^n$.

    We claim that
    $$\begin{array}{rrcl}
        \Psi: & \Hom_{K}(V,W) & \longrightarrow & V^\vee \otimes W\\
        & f & \longmapsto & \dis \sum_{i=1}^n e_i^\vee \otimes f(e_i)
    \end{array}$$
    is the inverse of $\Phi$.
    For any $(\varphi,w) \in V^\vee \times W$,
    $$\Psi(\Phi(\varphi \otimes w)) = \Psi (x \mapsto \varphi(x) w) = (\sum_{i=1}^n \varphi(e_i) e_i^\vee) \otimes w = \varphi \otimes w.$$
    For any $f \in \Hom_{K}(V,W)$,
    $$\begin{aligned}
        \Phi(\Psi(f)) = & \sum_{i=1}^{n} \Phi(e_{i}^\vee \otimes f(e_i)) = \sum_{i=1}^{n}(x \mapsto e_i^\vee (x)f(e_i))\\
        = & \left(x \mapsto f\left(\sum_{i=1}^{n}e_i^\vee (x)e_i\right)\right) = f.
    \end{aligned}$$
\end{proofenv}

\begin{definitionenv}
    Let $U \subseteq E$ be an open subset, $p \in \NN$.
    Let $\alpha : U \rightarrow \bigwedge^p E^*$ be a differentiable $p$-form.
    Then
    $$\DD \alpha : U \longrightarrow \Hom_{\RR} (E,\bigwedge^p E^*) \cong E^* \otimes \bigwedge^p E^*$$
    can be considered as a mapping from $U$ to $E^* \otimes \bigwedge^p E^*$.
    We denote by $\dd \alpha$ the mapping
    $$\begin{array}{rcccl}
        U & \overset{\DD \alpha}{\longrightarrow} & E^* \otimes \bigwedge^p E^* & \longrightarrow & \bigwedge^{p+1} E^*\\
        & & \varphi \otimes \beta & \longmapsto & \varphi \wedge \beta.
    \end{array}$$
\end{definitionenv}

\begin{exampleenv}
    If $f : U \rightarrow \RR$ is differentiable, then $\DD f : U \rightarrow E^*$, and $\dd f = \DD f$.
    If $p \in \NN$, $\omega : U \rightarrow \bigwedge^p E^*$ is a $p$-form,
    taking a basis $(x_i)_{i=1}^n$ of $E^*$, we can express $\omega$ as
    $$(x \in U) \longmapsto \sum_{1 \le i_1 \le \cdots \le i_p \le n} f_{i_1 \cdots i_p} (x) \dd x_{i_1} \wedge \cdots \wedge \dd x_{i_p}.$$
\end{exampleenv}

\begin{remark}
    In the study of $p$-forms, we may often assume, by linearity that a $p$-form $\omega$ is of the form $f\cdot\eta$, where $f : U \rightarrow \RR$ is a mapping and $\eta \in \bigwedge^{p} E^*$.
    If $f$ is differentiable,
    $$\dd(f \eta) = \dd f \wedge \eta.$$
\end{remark}

\begin{propositionenv}
    Let $p$ and $q$ be natural numbers. $U \subseteq E$ be an open subset, $\alpha : U \rightarrow \bigwedge^p E^*$ and $\beta : U \rightarrow \bigwedge^q E^*$ be differentiable mappings.
    Then $\alpha \wedge \beta$ is differentiable, and
    $$\dd(\alpha \wedge \beta) = \dd \alpha \wedge \beta + (-1)^p \alpha \wedge \dd \beta.$$
\end{propositionenv}
\begin{proofenv}
    We may assume that $\alpha = f \omega$, $\beta = g \eta$, where $f: U \rightarrow \RR$, $g: U \rightarrow \RR$ are differentiable and $(\omega,\eta) \in (\bigwedge^p E^*)\times (\bigwedge^q E^*)$.
    $$\dd \alpha = \dd f \wedge \omega,\quad \dd \beta = \dd g \wedge \eta.$$
    $$\dd \alpha \wedge \beta = \dd f \wedge \omega \wedge g \eta = g \dd f \wedge (\omega \wedge \eta)$$
    $$\alpha \wedge \dd \beta = f \omega \wedge (\dd g \wedge \eta) = f (-1)^p \dd g \wedge \omega \wedge \eta = (-1)^p f \dd g \wedge (\omega \wedge \eta).$$
    $$\dd \alpha \wedge \beta + (-1)^p \alpha \wedge \dd \beta = (g \dd f + f \dd g)\wedge (\omega \wedge \eta) = \dd(fg)\wedge (\omega \wedge \eta) = \dd (\alpha \wedge \beta).$$
\end{proofenv}

\begin{propositionenv}
    Let $K$ be a commutative unitary ring, $n \in \NN_{\ge 1}$ and $(V_{i})_{i=1}^n$ be a family of $K$-modules.
    There is a homomorphism of $K$-modules,
    $$\begin{array}{rrcl}
        \Phi : & V_1^\vee \otimes \cdots \otimes V_n^\vee & \longrightarrow & (V_1 \otimes \cdots \otimes V_n)^\vee\\
        & \varphi_1 \otimes \cdots \otimes \varphi_n & \longmapsto & (x_1 \otimes \cdots \otimes x_n \mapsto \varphi_1(x_1) \cdots \varphi_n(x_n)).
    \end{array}$$
    It is an isomorphism when $V_1,\cdots,V_n$ are free $K$-modules of finite rank.
\end{propositionenv}
\begin{proofenv}
    $$\begin{aligned}
        V_1^\vee \otimes \cdots \otimes V_n^\vee \cong & \Hom_{K} (V_1,V_2^\vee \otimes \cdots \otimes V_n^\vee) \cong \cdots\\
        \cong & \Hom_{K}\left(V_1,\Hom_{K}\left(V_2,\cdots \Hom_{K}(V_n,K)\right)\right)\\
        \cong &  \Hom_{K}^{(n)} (V_1\times \cdots \times V_n,K)\\
        = & \Hom_{K}(V_1 \otimes \cdots \otimes V_n, K) = (V_1\otimes \cdots \otimes V_n)^\vee.
    \end{aligned}$$
\end{proofenv}

\begin{definitionenv}
    Let $K$ be a commutative unitary ring, $n \in \NN$, $V$ be a $K$-module.
    Let
    $$\begin{array}{rrcl}
        \pi : & V^n & \longrightarrow & V^{\otimes n}\\
        & (x_1,\cdots,x_n) & \longmapsto & x_1 \otimes \cdots \otimes x_n
    \end{array}$$
    Note that $\mathfrak{S}_n$ acts from left to $V^n$ by permuting the coordinates.
    For any $\sigma \in \mathfrak{S}_n$, $(x_1,\cdots,x_n) \in V^n$, this action sends
    $(\sigma,(x_1,\cdots,x_n)) \mapsto (x_{\sigma^{-1}(1)},\cdots,x_{\sigma^{-1}(n)})$.

    \quad The mapping $(x_1,\cdots,x_n) \mapsto (x_{\sigma^{-1}(1)},\cdots,x_{\sigma^{-1}(n)})$ is $n$-linear, hence it induces a $K$-module homomorphism
    $$\begin{array}{rcl}
        V^{\otimes n} & \longrightarrow & V^{\otimes n}\\
        x_1 \otimes \cdots \otimes x_n & \longmapsto & x_{\sigma^{-1}(1)} \otimes \cdots \otimes x_{\sigma^{-1}(n)}
    \end{array}$$
    Thus we obtain a left action of $\mathfrak{S}_n$ on $V^{\otimes n}$.
    If $T \in V^{\otimes n}$ is such that $\forall \sigma \in \mathfrak{S}_n$, $\sigma(T) = T$, we say that $T$ is symmetric.
\end{definitionenv}

\begin{exampleenv}
    Let $(x_1,\cdots,x_n) \in V^n$, then
    $$T = \sum_{\sigma \in \mathfrak{S}_n} x_{\sigma(1)} \otimes \cdots \otimes x_{\sigma(n)}$$
    is symmetric.
\end{exampleenv}

\begin{propositionenv}
    Let $K$ be a commutative unitary ring, $n \in \NN$, $V$ be a $K$-module.
    For any $\varphi \in (V^\vee)^{\otimes n}$,
    let $\lambda_\varphi : V^n \rightarrow K$ be an $n$-linear mapping corresponding to $\Phi(\varphi)$,
    where $\Phi : (V^\vee)^{\otimes n} \rightarrow (V^{\otimes n})^\vee$ is the mapping that sends $\mu_1 \otimes \cdots \otimes \mu_n$ to $(x_1 \otimes \cdots \otimes x_n \mapsto \prod_{i=1}^{n}\mu_{i}(x_i))$.
    If $\varphi$ is symmetric, $\forall \sigma \in \mathfrak{S}_{n}$, $\forall (x_1,\cdots,x_n) \in V^n$, one has
    $$\lambda_\varphi (x_{\sigma(1)},\cdots,x_{\sigma(n)}) = \lambda_\varphi(x_1,\cdots,x_n).$$
    The converse is true when $V$ is free of finite rank.
\end{propositionenv}
\begin{proofenv}
    Note that $\varphi$ can be written as
    $$\sum_{i=1}^{n} \varphi_{i,1} \otimes \cdots \otimes \varphi_{i,n},\; \varphi_{i,j} \in V^\vee,$$
    $$\lambda_{\varphi}(x_1,\cdots,x_n) = \sum_{i=1}^{n} \varphi_{i,1}(x_1)\cdots \varphi_{i,n}(x_n).$$
    If $\varphi$ is symmetric, then $\forall \sigma \in \mathfrak{S}_n$,
    $$\begin{aligned}
        \varphi = & \sum_{i=1}^{n}\varphi_{i,\sigma^{-1}(1)} \otimes \cdots \otimes \varphi_{i,\sigma^{-1}(n)}(x_n) = \sum_{i=1}^{n}\varphi_{i,1}(x_{\sigma(1)}) \cdots \varphi_{i,n}(x_{\sigma(n)})\\
        = & \lambda_{\varphi}(x_{\sigma(1)},\cdots,x_{\sigma(n)})
    \end{aligned}.$$
    Assume that $(e_i)_{i=1}^r$ is a basis of $V$, $(e_j^\vee)_{j=1}^r$ be its dual basis.
    $$\varphi = \sum_{(j_1,\cdots,j_n) \in \{1,\cdots,r\}^n} \lambda_{\varphi} (e_{j_1},\cdots,e_{j_n})\cdot e_{j_1}^\vee \otimes \cdots \otimes e_{j_n}^\vee.$$
    $$\begin{aligned}
        \sigma(\varphi) = & \sum_{(j_1,\cdots,j_n) \in \{1,\cdots,r\}^n} \lambda_{\varphi} (e_{j_1},\cdots,e_{j_n})\cdot e_{j_{\sigma^{-1}(1)}}^\vee \otimes \cdots \otimes e_{j_{\sigma^{-1}(n)}}^\vee\\
        = & \sum_{(j_1,\cdots,j_n) \in \{1,\cdots,r\}^n} \lambda_{\varphi} (e_{j_{\sigma(1)}},\cdots,e_{j_{\sigma(n)}})\cdot e_{j_1}^\vee \otimes \cdots \otimes e_{j_n}^\vee
    \end{aligned}$$
    If for any $\sigma \in \mathfrak{S}_n$,
    $$\lambda_{\varphi}(x_1,\cdots,x_n) = \lambda_{\varphi}(x_{\sigma(1)},\cdots,x_{\sigma(n)}),$$
    then $\sigma(\varphi) = \varphi$.
\end{proofenv}

\begin{propositionenv}
    Let $K$ be a commutative unitary ring, $n \in \NN$, $V$ be a $K$-module, $\varphi \in V^{\otimes n}$.
    Suppose that $2 \coloneq 1+1$ is invertible in $K$.
    If $\varphi$ is symmetric, then the image of $\varphi$ by
    $$\begin{array}{rrcl}
        \pi : & V^{\otimes n} & \longrightarrow & \bigwedge^n V\\
        & x_1 \otimes \cdots \otimes x_n & \longmapsto & x_1 \wedge \cdots \wedge x_n
    \end{array}$$
    is zero.
\end{propositionenv}
\begin{proofenv}
    Let $\sigma = (1\;2)$, $\sigma(\varphi) = \varphi$.
    However
    $$\pi(\sigma(\varphi)) = - \pi(\varphi) = \pi(\varphi).$$
    $2 \pi(\varphi) = 0$ since $2$ is invertible.
\end{proofenv}

\begin{propositionenv}
    Let $p \in \NN$, $U \subseteq E$ be open $\alpha : U \rightarrow \bigwedge^p E^*$ be of class $C^2$.
    Then
    $$\dd^2 \alpha \coloneq \dd(\dd \alpha) = 0.$$
\end{propositionenv}
\begin{proofenv}
    We may assume $\alpha = f \cdot \eta$, $\eta \in \bigwedge^p E^*$, $f : U \rightarrow \RR$ is of class $C^2$.
    Then $\dd \alpha = \DD f \wedge \eta$.
    So $\dd^2 \alpha = \dd (\DD f) \wedge \eta$.
    For any $x \in U$, $\dd(\DD f)$ is the image of $\DD^2 f(x)$ in $\bigwedge^2 E^*$.
    Since $\DD^2f (x)$ is symmetric, one has $\dd(\DD f) = 0$.
    So $\dd^2 \alpha = 0$.
\end{proofenv}

\begin{definitionenv}
    We fix a non-zero element $\xi \in \det(E) \coloneq \bigwedge^n (E)$.
    Note that 
    $$\xi = e_1 \wedge \cdots \wedge e_n,$$
    where $(e_i)_{i=1}^{n}$, $n = \dim(E)$ is a basis of $E$ over $\RR$.
    Let $\mu_{\xi}$ be a Haar measure on $E$ such that the measure of the parallellotope
    $$\{a_1 e_1 + \cdots + a_n e_n \mid (a_1,\cdots,a_n) \in \interval[open left]{0}{1}^n\}$$
    is equal to $1$.

    \quad By theorem $\ref{thm:5.6.9}$, the Haar measure $\mu_{\xi}$ does not depend on the choice of the basis $(e_i)_{i=1}^{n}$.
    It only depends on the choice of the non-zero element $\xi \in \det(E)$,
    such that $\xi = e_1 \wedge \cdots \wedge e_n$. 

    \quad Moreover, for any non-zero real number $b$, one has
    $$\mu_{b \xi} = |b|^{-1} \mu_{\xi}.$$
    Let $U$ be an open subset of $E$ and $\alpha : U \longrightarrow \bigwedge^n(E^*)$.
    If the mapping
    $$(x \in U) \longmapsto \alpha(x)(\xi) \in \RR$$
    is integrable with respect to the Haar measure $\mu_{\xi}$,
    then we say that $\alpha$ is integrable and we denote by 
    $$\int_{U} \alpha$$
    the integral
    $$\int_{U} \alpha(x) (\xi) \mu_{\xi}(\dd x).$$
    Note that the integral only depends on the orbit of $\xi$ in $\det(E)$ by the action of $\RR_{>0}$ which is called an orientation of $E$.

    \quad Let $p$ be an element of the set $\{0,\cdots,n\}$ and let $\alpha : U \rightarrow \bigwedge^p(E^*)$ be a $p$-form.
    If for any $\beta \in \Omega_{c}^{n-p}(U)$, $\alpha \wedge \beta$ is an integrable $n$-form, we say that $\alpha$ is locally integrable.
    We denote by $\Omega_{\text{loc}}^{p}(U)$ the set of all locally integrable $p$-forms on $U$.
\end{definitionenv}


\section{Localization and Mollification}


In this section, we fix a finite-dimensional vector space $E$ over $\RR$.

\begin{lemmaenv}
    The mapping
    $$\begin{array}{rrcl}
        b : & \RR & \longrightarrow & [0,1]\\
        & t & \longmapsto & \begin{cases}
            \exp\left(\frac{1}{t^2 - 1}\right), & |t| < 1\\
            0, & |t|\ge 1
        \end{cases}
    \end{array}$$
    is smooth.
\end{lemmaenv}
\begin{proofenv}
    We first prove that for any natural number $n$,
    $$\lim_{x \to +\infty} x^n \exp(-x) = 0.$$
    In fact, by the equality
    $$\exp(x) = \sum_{k \in \NN} \frac{x^k}{k!},$$
    we obtain that
    $$\exp(x) \ge \frac{x^{n+1}}{(n+1)!},$$
    and hence
    $$\forall x \in \RR_{>0},\; 0 \le x^n \exp(-x) \le \frac{(n+1)!}{x}.$$
    Taking the limit when $x \rightarrow +\infty$, we get
    $$\lim_{x \to +\infty} x^n \exp(-x) = 0.$$
    
    \quad Now we prove by induction that the mapping $b$ is differentiable $n$ times and 
    $$b^{(n)}(t) = \begin{cases}
        \frac{F_n(t)}{(t^2 - 1)^{2n}} b(t), & |t| < 1\\
        0, & |t|\ge 1
    \end{cases},$$
    where $F_n$ is a polynomial.
    By definition, the statement is true when $n = 0$.
    We assume that the statement is true for $n$.
    Then the function $b^{(n)}$ is differentiable on $\interval[open]{-1}{1}$ and on this interval
    $$\begin{aligned}
        b^{(n+1)}(t) = & \frac{F_{n}(t)}{(t^2 - 1)^{2n}}b'(t) - \frac{4ntF_n(t)}{(t^2 - 1)^{2n+1}}b(t) + \frac{F'_n(t)}{(t^2 - 1)^{2n}} b(t)\\
        = & \frac{-2t F_n(t) - 4nt (t^2 - 1) F_n(t) +(t^2 - 1)^2 F'_n(t)}{(t^2 - 1)^{2n+2}} b(t).
    \end{aligned}$$
    Moreover, $b^{(n)}$ is also differentiable on $\RR\setminus\interval{-1}{1}$ and $b^{(n+1)}(t) = 0$ for any $t \in \RR\setminus[-1,1]$.

    \quad By the change of variable $x = - \frac{1}{t^2 - 1}$, we obtain
    $$\lim_{|t| \to 1^-} \frac{b(t)}{(t^2 - 1)^{2n + 1}} = \lim_{x \rightarrow +\infty} (-1)^{2n+1} x^{2n + 1} \exp(-x) = 0.$$
    As a consequence, we obtain that
    $$\lim_{t \rightarrow 1} \frac{b^{(n)}(t)}{t - 1} = \lim_{t \rightarrow -1} \frac{b^{(n)}(t)}{t + 1} = 0.$$
    In particular, $b^{(n)}$ is differentiable at $1,-1$ and 
    $$b^{(n+1)}(1) = b^{(n+1)}(-1) = 0.$$
\end{proofenv}

\begin{propositionenv}
    Let $U$ be an open subset of $E$ and $K \subseteq U$ be compact.
    There exists a smooth mapping
    $$\rho : E \longrightarrow [0,1]$$
    such that $\rho(x) = 1$ for any $x \in K$ and $\supp(\rho) \subseteq U$.
\end{propositionenv}
\begin{proofenv}
    Without loss of generality, we may assume that $E = \RR^n$, $n \in \ZZ_{>0}$.
    We equip $\RR^n$ with the norm $\|\cdot\|$ such that
    $$\|(x_1,\cdots,x_n)\| = \max_{i \in \{1,\cdots,n\}} |x_i|.$$
    Let $\chi : \RR^n \rightarrow \RR$, $\chi(x_1,\cdots,x_n) = C b(x_1) \cdots b(x_n)$,
    where $C$ is a positive constant such that
    $$C\left(\int_{\RR} b(t) \dd t\right)^n = 1.$$
    Note that by Fubini's theorem,
    $$\int_{\RR^n} \chi(x_1,\cdots,x_n)\dd(x_1,\cdots,x_n) = 1$$
    where $\dd (x_1,\cdots,x_n)$ denotes the Lebesgue measure on $\RR^n$.
    For any $\varepsilon \in \RR_{>0}$, we denote by $\chi_{\varepsilon}$ the mapping
    $\chi_{\varepsilon} : E \rightarrow \RR$ defined as $\chi_{\varepsilon}(x) \coloneq \frac{1}{\varepsilon^n} \chi(x/\varepsilon)$.
    By the formula of change of variables
    $$\int_{\RR^n} \chi_{\varepsilon}(x_1,\cdots,x_n) \dd(x_1,\cdots,x_n) = 1.$$
    For any $\varepsilon \in \RR_{>0}$, define $K_{\varepsilon} = \bigcup_{y \in K}\bar{B}(y,\varepsilon)$.
    Clearly, for any $x \in K_{\varepsilon}$, one has
    $$\dd(x,K) \coloneq \inf_{y \in K} \dd(x,y) \le \varepsilon.$$

    \quad Conversely, if $x$ is an element of $E$ such that $\dd(x,K)\le \varepsilon$, then there exists a sequence $(y_n)_{n \in \NN}$ of elements in $K$ such that
    $$\lim_{n\rightarrow +\infty}\|x - y_n\|\le \varepsilon.$$
    By taking a subsequence, we may assume without loss of generality that $(y_n)_{n \in \NN}$ converges to some $y \in K$.
    Hence
    $$\dd(x,y) \le \varepsilon.$$
    We obtain that
    $$K_{\varepsilon} = \{x \in E \mid \dd(x,K) \le \varepsilon\}$$
    is a compact subset of $E$ (closed and bounded).
    
    \quad We claim that there exists $\varepsilon > 0$ such that
    $$K_{2\varepsilon} \subseteq U.$$
    Let $(\varepsilon_n)_{n \in \NN}$ be a decreasing sequence converging to $0$.
    Suppose by contradiction that for any $n \in \NN$, there exists $x_n \in K_{\varepsilon_n} \setminus U$.
    Since $K_{\varepsilon_0}$ is compact,
    by taking a subsequence of $(x_n)_{n \in \NN}$ we may assume without loss of generality that $(x_n)_{n \in \NN}$ converges to some $x \in K_{\varepsilon_0}$.
    Since $U$ is open, $x \notin U$.
    However, one has $\dd(x,K) = 0$ and hence $x \in K \subseteq U$ (contradiction).
    
    \quad We now let $\rho$ to be the convolution of $\mathbbm{1}_{K_{\varepsilon}}$ and $\chi_{\varepsilon}$, i.e.
    $$\forall x \in \RR^n,\; \rho(x) = \int_{\RR^n} \mathbbm{1}_{K_{\varepsilon}}(y)\chi_{\varepsilon}(x-y)\dd y.$$
    Since $\chi_{\varepsilon}$ is smooth and of compact support, by the theorem of differentiability under integral,
    we get that $\rho$ is smooth.
    By the formula of change of variable and the fact that $\chi_{\varepsilon}$ is an even function
    $$\rho(x) = \int_{\RR^n} \mathbbm{1}_{K_{\varepsilon}} (x - y)\chi_{\varepsilon} (y) \dd y.$$
    For any $x \in K$,
    $$\rho(x) = \int_{\RR^n} \chi_{\varepsilon}(y) \dd y = 1.$$
    For any $x \notin K_{2\varepsilon}$, one has
    $$\rho(x) = \int_{K_{2\varepsilon}} \mathbbm{1}_{K_{\varepsilon}}(x+y)\chi_{\varepsilon}(y) \dd y = 0.$$
    Hence the support of $\rho$ is contained in $K_{2\varepsilon} \subseteq U$.
    In particular $\supp(\rho) \subseteq U$.
\end{proofenv}

\begin{remark}
    Let $U$ be an open subset of $E$, $K \subseteq U$ be compact and $f : U \rightarrow \RR$ be a mapping.
    We pick a smooth mapping $\rho : E \rightarrow \RR$ of compact support such that $\rho(x) = 1$ for any $x \in K$,
    and $\supp(\rho) \subseteq U$.
    Then the mapping
    $$(x \in E) \longmapsto \begin{cases}
        f(x) \rho(x), & x \in U\\
        0, & x \in E \setminus U
    \end{cases}$$
    coincides with $f$ on $K$ and is of compact support contained in $U$ of class $C^l$ if $f$ is of class $C^l$.
\end{remark}

\begin{lemmaenv}\label{lem:7.2.4}
    Let $\Omega$ be a set, $S \subseteq \RR^\Omega$ be a Riesz space and $I : S \rightarrow \RR$ be an integral operator.
    Then $S$ is dense in $\mathcal{L}^{1}(I)$,
    namely, for any $f \in \mathcal{L}^{1}(I)$ and any $\varepsilon > 0$,
    there exists $g \in S$ such that $I(|f - g|) \le \varepsilon$.
\end{lemmaenv}
\begin{proofenv}
    Without loss of generality, we may assume that $f$ is non-negative ($f = f^+ - f^-$).
    
    For any $\varepsilon > 0$, $u \in S^\uparrow$ such that $u \ge f$ and $I(u) - I(f) \le \frac{\varepsilon}{2}$.
    ($I(f) = \inf \{I(u), u\in S^\uparrow, u\ge f\}$).
    Let $(g_n)_{n \in \NN} \in S^\NN$ be an increasing sequence of elements in $S$ that converges pointwise to $u$.
    For any $n \in \NN$, let $f_n = \inf\{f, g_n\}$,
    one has $f_n \le g_n$ and $g_n - f_n \le u - f$.
    $$I(|g_n - f_n|) = I(g_n - f_n) \le I(u - f) \le \frac{\varepsilon}{2}.$$

    The sequence $(f - f_n)_{n \in \NN}$ is decreasing and converges pointwise to $0$.
    Since $I : \mathcal{L}^{1}(I) \rightarrow \RR$ is an integral operator,
    $$\lim_{n \rightarrow +\infty} I(|f - f_n|) = 0.$$
    Hence, there exists $n_0 \in \NN$ such that 
    $$I(|f - f_n|) = I(f - f_n) \le \frac{\varepsilon}{2}.$$
    Hence $I(|g_{n_0} - f|) \le \varepsilon$.
\end{proofenv}

\begin{propositionenv}
    We equip $E$ with a norm $\|\cdot\|$ and with a Haar measure $\mu$.
    Let $\chi : E \rightarrow [0,1]$ be a smooth mapping with support contained in
    $$B = \{x \in E \mid \|x\| \le 1\}$$
    such that
    $$\int_{E} \chi(x) \mu(\dd x) = 1.$$
    Let $f : E \rightarrow \RR$ be a mapping that is integrable with respect to the measure $\mu$.
    For any $\varepsilon > 0$, we let $\chi_{\varepsilon} : E \rightarrow \RR$ be the mapping defined as $\chi_{\varepsilon} = \frac{1}{\varepsilon^n}\chi(\frac{x}{\varepsilon})$,
    where $n = \dim_{\RR} E$.
    Let $f_\varepsilon$ be the convolution of $f$ and $\chi_{\varepsilon}$, i.e.
    $$f_\varepsilon(x) = \int_{E} f(x-y)\chi_{\varepsilon}(y)\dd y = \int_{E} f(y)\chi_{\varepsilon}(x-y)\dd y.$$
    \begin{enumerate}[label=(\arabic*)]
        \item Suppose that $f$ is continuous at $x_0 \in E$, then
            $$\lim_{\varepsilon \to 0} f_\varepsilon(x_0) = f(x_0).$$
        \item Let $U \subseteq E$ be open such that $f|_U$ is uniformly continuous.
            For any compact $K \subseteq U$, the family $(f_{\varepsilon})_{\varepsilon > 0}$ converges uniformly to $f$ on $K$.
        \item Suppose that $f$ is integrable on $E$, then
            $$\lim_{\varepsilon \to 0} \int_{E} |f_\varepsilon(x) - f(x)| \mu(\dd x) = 0.$$
    \end{enumerate}
\end{propositionenv}
\begin{proofenv}
    For any $\varepsilon > 0$, let $B_{\varepsilon} = \{x \in E \mid \|x\| \le \varepsilon\}$.
    \begin{enumerate}
        \item Since
            $$\int_E \chi_{\varepsilon}(y) \mu(\dd y) = 1.$$
            We obtain that
            $$\begin{aligned}
                |f_\varepsilon(x_0) - f(x_0)| = & \left|\int_E (f(x_0-y) - f(x_0))\chi_{\varepsilon}(y)\mu(\dd y)\right|\\
                \le & \int_E |f(x_0-y) - f(x_0)|\chi_{\varepsilon}(y)\mu(\dd y)\\
                \le & \sup_{y \in B_{\varepsilon}}|f(x_0 - y) - f(x_0)|.
            \end{aligned}$$
            Since $f$ is continuous at $x_0$, 
            $$\lim_{\varepsilon \to 0} \sup_{y \in B_{\varepsilon}}|f(x_0 - y) - f(x_0)| = 0,$$
            which leads to $\lim_{\varepsilon \to 0} f_\varepsilon(x_0) = f(x_0)$.
        \item Let $\varepsilon > 0$ be such that
            $$K_{\varepsilon} = \{x \in E \mid \dd(x,K) \le \varepsilon\} \subseteq U.$$
            By (1), for any $x \in K$, one has
            $$\sup_{x \in K} |f_{\varepsilon}(x) - f(x)| \le \sup_{x \in K} \sup_{y \in B_{\varepsilon}} |f(x-y) - f(x)| \le \sup_{\substack{a,b \in U\\ \|a-b\|\le \varepsilon}}|f(a) - f(b)|.$$
            Hence
            $$\lim_{\varepsilon \to 0} \sup_{x \in K}|f_{\varepsilon}(x) - f(x)| = 0.$$
        \item We first consider the case where $f \in \cc(E)$.
            Let $K$ be the compact set such that $\supp(f) = K$.

            \quad Note that $f$ is uniformly continuous on $E$.
            Let $K_1 = \{y \in E \mid \dd(y,K) \le 1\}$.
            For any $0 < \varepsilon < 1$, the support of $f_{\varepsilon}$ is contained in the $K_1$.
            By (2),
            $$\lim_{\varepsilon \rightarrow 0} \sup_{x \in E} |f_{\varepsilon}(x) - f(x)| = 0.$$
            Since for any $\varepsilon \in \interval[open]{0}{1}$,
            $$0 \le \int_{E} |f_{\varepsilon}(x) - f(x)| \mu(\dd x) \le \mu(K_1) \sup_{x \in E}|f_{\varepsilon}(x) - f(x)|$$
            we obtain
            $$\lim_{\varepsilon \rightarrow 0} \int_{E} |f_{\varepsilon} - f(x)|\mu(\dd x) = 0.$$
            
            \quad We now consider the general case.
            Let $\delta > 0$, by lemma \ref{lem:7.2.4}, there exists $g \in \cc(E)$ such that 
            $$\int_{E} |f(x) - g(x)|\mu(\dd x) \le \delta,$$
            $$\begin{aligned}
                \int_{E} |f_{\varepsilon}(x) - g_{\varepsilon}(x)| \mu(\dd x) = & \int_{E} \left|\int_{E} (f(y) - g(y))\chi_{\varepsilon}(x - y)\dd y\right| \mu(\dd x)\\
                \le & \int_{E} \int_{E} |f(y) - g(y)|\chi_{\varepsilon}(x-y)\mu(\dd y) \mu(\dd x)\\
                = & \int_{E} \int_{E} |f(y) - g(y)|\chi_{\varepsilon}(x-y)\mu(\dd x) \mu(\dd y)\\
                = & \int_{E} |f(y) - g(y)| \mu(\dd y) \le \delta.
            \end{aligned}$$
            By the triangle inequality applied to
            $$f_{\varepsilon} - f = f_{\varepsilon} - g_{\varepsilon} + g_{\varepsilon} - g + g - f$$
            we get that
            $$\int_{E} |f_{\varepsilon}(x) - f(x)| \mu(\dd x) \le \int_{E} |g_{\varepsilon}(x) - g(x)| \mu(\dd x) + 2 \delta,$$
            which leads to
            $$\limsup_{\varepsilon \to 0} \int_{E} |f_{\varepsilon}(x) - f(x)| \mu(\dd x) \le \delta.$$
            Since $\delta$ is arbitrary, we obtain that
            $$\lim_{\varepsilon \to 0} \int_{E} |f_{\varepsilon}(x) - f(x)| \mu(\dd x) = 0.$$
    \end{enumerate}
\end{proofenv}

\begin{theoremenv}[Taylor's formula]\label{thm:7.2.6}
    Let $(E,\|\cdot\|)$ be a Banach space, $U \subseteq E$ be open,
    $a \in U$ and $h \in E$.
    Suppose that for any $t \in [0,1]$, $a + th \in U$.
    Let $f : U \rightarrow \RR$ be a mapping of class $C^{n+1}$.
    The following equality holds:
    $$f(a + h) = \sum_{k=0}^{n} \frac{1}{k!} \DD^k f(a)(h,\cdots,h) + \int_{0}^{1}\frac{(1 - t)^n}{n!} \DD^{n+1}f(a + th)(h,\cdots,h)\dd t.$$
\end{theoremenv}
\begin{proofenv}
    We reason by induction on $n$.
    In the case where $n = 0$, one has
    $$\frac{\dd}{\dd t} f(a + th) = \DD f(a + th)(h),$$
    so that
    $$\int_{0}^{1} \DD f(a + th)(h) \dd t = f(a + h) - f(a),$$
    by the fundamental theorem of analysis.

    \quad Suppose that the formula holds for $n-1$.
    By the relation
    $$\frac{\dd}{\dd t}\DD^n f(a + th)(h,\cdots,h) = \DD^{n+1}f(a + th)(h,\cdots,h),$$
    Integrating by parts, we obtain
    $$\begin{aligned}
        & \int_{0}^{1} \frac{(1 - t)^n}{n!} \DD^{n+1}f(a + th)(h,\cdots,h) \dd t\\
        = & -\frac{1}{n!}\DD^n f(a)(h,\cdots,h) + \int_{0}^{1} \frac{(1 - t)^{n-1}}{(n-1)!} \DD^n f(a + th)(h,\cdots,h) \dd t.
    \end{aligned}$$
    By inductive assumption, we obtain the desired formula.
\end{proofenv}

\begin{remark}
    Let $E$ be a finite-dimensional vector space over $\RR$,
    $U \subseteq E$ be open and convex.
    Let $a \in U$ and $f : U \rightarrow \RR$ be a mapping of class $C^{l}$, $l \in \NN_{\ge 1} \cup \{\infty\}$.
    By theorem \ref{thm:7.2.6},
    $$f(a + h) = f(a) + \int_{0}^{1} \DD f(a + th)(h) \dd t.$$
    Let $(e_i)_{i=1}^{n}$ be a basis of $E$ and $(e_i^\vee)_{i=1}^{n}$ be the dual basis.
    One has
    $$f(a + h) = f(a) + \sum_{i=1}^{n} e_i^\vee (h) \int_{0}^{1} \DD f(a + th)(e_i) \dd t.$$
    In other words, there exist continuous mappings $g_1,\cdots,g_n : U \rightarrow \RR$, $i \in \{1,\cdots,n\}$ such that
    $$\forall x \in U,\; f(x) = f(a) + \sum_{i=1}^{n} g_i(x) e^\vee_i (x - a)$$
    where
    $$g_i(x) = \int_{0}^{1} \DD f(tx + (1-t)a)(e_i) \dd t.$$
\end{remark}

\begin{definitionenv}
    Let $K$ be a commutative unitary ring,
    $A$ be a commutative algebra.
    Let $P : A \rightarrow K$ be a homomorphism of $K$-algebras.
    For any $f \in A$, we denote by $f(P)$ the image of $f$ in $K$ by the homomorphism $P$.
    We call \textbf{point derivation} at $P$ any homomorphism of $K$-modules $\partial : A \rightarrow K$ which satisfies the following Leibniz rule:
    $$\forall (f,g) \in A^2,\; \partial(fg) = f(P) \partial(g) + g(P) \partial(f).$$
    We denote by $\Der_P(A)$ the set of all point derivations at $P$ of the $K$-algebra $A$.
    It forms a sub-$K$-module of $K^A$.

    \quad For any $c \in K$, we define $\partial(c)$ via the identification $c \mapsto c\cdot 1_A$.
\end{definitionenv}

\begin{exampleenv}
    Let $(E,\|\cdot\|)$ be a Banach space over $\RR$, $\varnothing \neq U \subseteq E$ open, $P \in U$.
    Let $l \in \NN_{\ge 1} \cup \{\infty\}$.
    By abuse of notation, we denote by $P$ the evaluation mapping
    $$P : C^l(U) \ni f \longmapsto f(P) \in \RR.$$
    This is a homomorphism of $\RR$-algebras.
    For any $v \in E$, the mapping
    $$\left(f \in C^l(U)\right) \mapsto \DD f(P)(v)$$
    is a point derivation at $P$.
\end{exampleenv}

\begin{propositionenv}
    Let $K$ be a commutative unitary ring, $A$ be a commutative $K$-algebra and $P : A \rightarrow K$ be a homomorphism of $K$-algebras.
    Let $I = \ker(P)$.
    \begin{enumerate}[label=(\arabic*)]
        \item Let $\partial : A \rightarrow K$ be a derivation at $P$.
            For any $f \in I^2$, $\partial(f) = 0$.
            Hence $\partial$ induces a homomorphism of $K$-modules $I/I^2 \rightarrow K$.
        \item  Suppose that $2$ is not a zero divisor in $K$.
            For any $c \in K$, one has $\partial(c) = 0$.
        \item Suppose that $2$ is not a zero divisor in $K$.
            The mapping 
            $$\begin{array}{rrcl}
                \Phi : & \Der_P(A) & \longrightarrow & \Hom_K(I/I^2, K)\\
                & \partial & \longmapsto & ([f] \mapsto \partial(f))
            \end{array}$$
            is an isomorphism of $K$-modules.
    \end{enumerate}
\end{propositionenv}
\begin{proofenv}
    \quad 
    \begin{enumerate}[label=(\arabic*)]
        \item By definition, any $u \in I^2$ can be written as
            $$u = \sum_{i=1}^{n} f_i g_i,\;f_i,g_i \in I,i\in\{1,\cdots,n\}.$$
            In particular,
            $$f_1(P) = \cdots = f_n(P) = g_1(P) = \cdots = g_n(P) = 0.$$
            By Leibniz law
            $$\partial(u) = \sum_{i=1}^{n} \partial(f_i g_i) = \sum_{i=1}^{n} f_i(P) \partial(g_i) + g_i(P) \partial(f_i) = 0.$$
        \item By definition, one has 
            $$\partial(1) = \partial(1^2) = \partial(1) + \partial(1) = 2 \partial(1) = 0.$$
        \item For any $K$-linear mapping $\lambda : I/I^2 \rightarrow K$, we let $\Psi(\lambda)$ be the mapping
            $$\begin{array}{rrcl}
                \Psi(\lambda) : & A & \longrightarrow & K\\
                & f & \longmapsto & \lambda([f - f(P)])
            \end{array}$$
            We claim that $\Psi(\lambda) \in \Der_P(A)$ and that
            $$\Psi : \Hom_{K}(I/I^2, K) \longrightarrow \Der_P(A)$$
            is the inverse of $\Phi$.
            
            \quad Let $f,g \in A$, $(f-f(P))(g-g(P)) \in I^2$ and
            $$\begin{aligned}
                [fg - f(P)g(P)] = & [f(P)(g - g(P)) + g(P)(f-f(P))]\\
                = & f(P)[g - g(P)] + g(P)[f - f(P)]
            \end{aligned}$$
            in $I/I^2$.
            Hence
            $$\Psi(\lambda)(fg) = f(P)\Psi(\lambda)(g) + g(P)\Psi(\lambda)(f).$$
            Finally, for any $\partial \in \Der_P(A)$, by (2),
            $$\Psi\left(\Phi(\partial)\right)(f) = \Phi(\partial)([f - f(P)]) = \partial(f - f(P)) = \partial(f)$$
            for any $\lambda \in \Hom_{K}(I/I^2, K)$ one has
            $$\Phi(\Psi(\lambda))([f]) = \Psi(\lambda)([f]) = \lambda([f - f(P)]) = \lambda([f]).$$
            since $f(P) = 0$.
    \end{enumerate}
\end{proofenv}

\begin{propositionenv}
    Let $E$ be a finite-dimensional vector space over $\RR$, $U \subseteq E$, $p \in U$.
    Let $\partial : C^l(U) \rightarrow \RR$ be a point derivation at $p$.
    Then $\partial$ is local at $p$, namely, if $f,g \in C^l(U)$ coincide on a neighborhood of $U$, 
    then $\partial(f) = \partial(g)$.
\end{propositionenv}

\begin{proofenv}
    It suffices to show that if $f \in C^l(U)$ vanishes on an open neighborhood $W$ of $p$ in $U$, then $\partial(f) = 0$.

    \quad Let $K$ be a compact neighborhood of $p$ such that $K \subseteq U$.
    Let $\lambda : U \rightarrow [0,1]$ of class $C^\infty$ such that $\lambda|_K \equiv 1$ and $\supp(\lambda) \subseteq W$, $\lambda f \equiv 0$.
    $$0 = \partial(\lambda f) = \lambda(p) \partial(f) + f(p) \partial(\lambda) = \partial(f).$$
\end{proofenv}

\begin{propositionenv}
    Let $E$ be a finite-dimensional vector space over $\RR$, $U \subseteq E$ open, $p \in U$.
    Let $l \in \NN_{\ge 1} \cup \{\infty\}$.
    Let $I = \{f \in C^l(U) \mid f(p) = 0\}$.
    Then the mapping
    $$\begin{array}{rrcl}
        \Phi : & E^* & \longrightarrow & I/I^2\\
        & \varphi & \longmapsto & [(\varphi - \varphi(p))|_U]
    \end{array}$$
    is injective.
    It is a bijection when $l = \infty$ and $U$ is convex.
\end{propositionenv}
\begin{proofenv}
    For any $f \in I^2$, one has $\DD f(p) = 0$.
    Therefore,
    $$\begin{array}{rcl}
        I & \longrightarrow & E^*\\
        f & \longmapsto & \DD f(p)
    \end{array}$$
    induces by passing to quotient an $\RR$-linear mapping $\Psi : I/I^2 \longrightarrow E^*$.
    For any $\varphi \in E^*$, $\Phi(\varphi) = [\varphi - \varphi(p)]$, $\Psi(\Phi(\varphi)) = \DD(\varphi - \varphi(p))(p) = \varphi$.
    $\Psi \circ \Phi = \Id_{E^*}$, so $\Phi$ should be injective.
    
    \quad Now we assume that $l = \infty$ and $U$ is convex.
    Let $(e_i)_{i=1}^{n}$ be a basis of $E$ and $(\varphi_i)_{i=1}^{n}$ be the dual basis.
    By the Taylor's formula (integral form), for any $f \in I$, there exist $g_1,\cdots,g_n$ in $C^\infty(U)$ with $g_i(p) = \partial_{e_i} f(p)$ such that
    $$f = \sum_{i=1}^{n} g_i (\varphi_i - \varphi_i(p)).$$
    ($f \in I^2 \Leftrightarrow \DD f(p) = 0$.)
    Therefore, for any $f \in I$,
    $$f - \DD f(p) + \DD f(p)(p) \in I^2.$$
    Hence $[f] = [\DD f(p) - \DD f(p)(p)] \in \mathrm{Im} (\Phi)$, so $\Phi$ is surjective.
\end{proofenv}


\section{Vector Fields}


We fix a finite-dimensional vector space $E$ over $\RR$ and $U \subseteq E$ be an open subset.

\begin{definitionenv}
    We call a tangent vector field any mapping 
    $$X : U \longrightarrow E.$$
    Let $f : U \rightarrow \RR$ be a differentiable mapping.
    If $X : U \rightarrow E$ is a tangent vector field, we denote by 
    $$\begin{array}{rrcl}
        X(f) & : & U & \longrightarrow \RR \\
             &   & x & \longmapsto \DD f(x)(X(x)).
    \end{array}$$
    If $f$ is a constant, the $X(f) = 0$.
\end{definitionenv}

\begin{remark}
    \quad 
    \begin{enumerate}
        \item If $f$ and $g$ are differentiable mappings from $U$ to $\RR$, then 
            $$\DD(fg) = f \DD (g) + g \DD (f).$$
            So
            $$X(fg) = f X(g) + g X(f).$$
            In particular, $X$ is $\RR$-linear.
        \item Let $(e_i)_{i=1}^n$ be a basis of $E$ and $(\varphi_i)_{i=1}^n$ be the dual basis of $(e_i)_{i=1}^{n}$.
            For any $v \in E$,
            $$v = \sum_{i=1}^{n} \varphi_i (v) e_i.$$
            So any tangent vector field $X$ can be written as
            $$X = \sum_{i=1}^{n} X(\varphi_i) e_i.$$
            So $X$ is of class $C^l$ if and only if $X(\varphi_i)$ is of class $C^l$ for any $i \in \{1,\cdots,n\}$.
            The set of tangent vector fields of class $C^l$ forms a free $C^l(U)$-module generated by the constant vector fields $e_1,\cdots,e_n$.
            
            \quad Moreover, if $X$ and $Y$ are tangent vector fields such that $X(\varphi) = Y(\varphi)$ for any $\varphi \in E^*$, then $X = Y$.
   \end{enumerate}
\end{remark}

\begin{propositionenv}
    Let $X$ and $Y$ be a tangent vector fields of class $C^1$ on $U$.
    There exists a unique tangent vector field of class $C^0$, $[X,Y]$ such that, for any $h \in C^2(U)$,
    $$[X,Y](h) = X\left(Y(h)\right) - Y\left(X(h)\right).$$
\end{propositionenv}
\begin{proofenv}
    By (bi)linearity, we may assume that $X = f v$, and $Y = g w$, with $f,g \in C^1(U)$ and $v,w \in E$.
    For any $h \in C^2(U)$,
    $$\begin{aligned}
        X\left(Y(h)\right) = & f \partial_{v}(Y(h)) = f \partial_{v}(g\partial_{w}(h))\\
        = & f \partial_{v}(g) \partial_{w}(h) + fg \partial_{v}\partial_{w}(h).
    \end{aligned}$$
    $$\begin{aligned}
        Y\left(X(h)\right) = & g \partial_{w}(X(h)) = g \partial_{w}(f\partial_{v}(h))\\
        = & g \partial_{w}(f) \partial_{v}(h) + fg \partial_{w}\partial_{v}(h).
    \end{aligned}$$
    So
    $$\begin{aligned}
        X\left(Y(h)\right) - Y\left(X(h)\right) = & f\partial_{v}(g)\partial_{w}(h) - g\partial_{w}(f)\partial_{v}(h)\\
        = & \left(f \partial_{v} (g) w - g \partial_{w}(f) v\right)(h).
    \end{aligned}$$
\end{proofenv}

\begin{remark}
    \quad
    \begin{enumerate}
        \item $(X,Y) \rightarrow [X,Y]$ is $\RR$-bilinear.
        \item If $X$ and $Y$ are of class $C^m$ ($m \ge 1$), then $[X,Y]$ is of class $C^{m-1}$.
    \end{enumerate}
\end{remark}

\begin{definitionenv}
    Let $E$ and $E'$ be a finite-dimensional vector space over $\RR$, $U \subseteq E$, $U' \subseteq E'$ be open,
    and $\psi : U \rightarrow U'$ be a $C^1$ diffeomorphism.
    If $X$ is a tangent vector field on $U$, we denote by
    $$\begin{array}{rrcl}
        \psi_* (X): & U' & \longrightarrow & E'\\
                    & y  & \longmapsto     & \DD\psi\left(\psi^{-1}(y)\right)\left(X (\psi^{-1}(y))\right).
    \end{array}$$
\end{definitionenv}

\begin{propositionenv}
    Assume that $\psi : U \rightarrow U'$ is a $C^2$ diffeomorphism.
    If $X$ and $Y$ are tangent vector fields of class $C^1$, then 
    $$\psi_*\left([X,Y]\right) = [\psi_* X, \psi_* Y].$$
\end{propositionenv}
\begin{proofenv}
    Let $h \in C^2(U')$.
    If $Z$ is a vector field of class $C^1$ on $U$, for any $x \in U$,
    $$\begin{aligned}
        \psi_*(Z)(h)(\psi(x)) = & \DD h (\psi(x))\left(\psi_*(Z) (\psi(x))\right)\\
        = & \DD h (\psi(x))\DD \psi(x)(Z(x))\\
        = & \DD(h \circ \psi)(x)(Z(x))\\
        = & Z(h \circ \psi)(x).
    \end{aligned}$$
    So $$\psi_*(Z)(h) = Z(h \circ \psi) \circ \psi^{-1}.$$
    $$\begin{aligned}
        [\psi_* (X), \psi_* (Y)](h) = & \psi_*(X)(\psi_* (Y)(h)) - \psi_*(Y)(\psi_*(X)(h))\\
        = & \psi_*(X)(Y(h \circ \psi) \circ \psi^{-1}) - \psi_*(Y)(X(h \circ \psi) \circ \psi^{-1})
    \end{aligned}$$
    $$\begin{aligned}
        X(Y(h \circ \psi)) \circ \psi^{-1} - Y(X(h \circ \psi)) \circ \psi^{-1} = & [X,Y](h \circ \psi) \circ \psi^{-1}\\
        = & \psi_*([X,Y])(h).
    \end{aligned}$$
\end{proofenv}

Reminder of algebra:

Let $K$ be a commutative unitary ring and $V$ be a $K$-module.
$p \in \NN$.
For any $(\varphi_1,\cdots,\varphi_p) \in (V^\vee)^p$,
the mapping
$$\begin{array}{rcl}
    V^p & \longrightarrow & K\\
    (e_1,\cdots,e_p) & \longmapsto & \det \begin{pmatrix}
        \varphi_1(e_1) & \cdots & \varphi_1(e_p)\\
        \vdots & & \vdots\\
        \varphi_p(e_1) & \cdots & \varphi_p(e_p)
    \end{pmatrix}
\end{array}$$
is a alternating multilinear form, 
so it induces a linear form
$$\tilde{\Phi} (\varphi_1,\cdots,\varphi_p) : \bigwedge^p (V) \longrightarrow K.$$
The mapping $\tilde{\Phi}: (V^\vee)^p \longrightarrow (\bigwedge^p (V))^\vee$ is an alternating multilinear mapping.
Hence it induces a $K$-linear mapping
$$\Phi : \bigwedge^p (V^\vee) \longrightarrow (\bigwedge^p (V))^\vee.$$
If $\alpha \in \bigwedge^p(V^\vee)$, we denote by $\alpha (x_1,\cdots,x_p)$ the element $\Phi(\alpha)(x_1 \wedge \cdots \wedge x_p)$.
If $\theta \in V^\vee$ and $\beta \in \bigwedge^p (V^\vee)$, then $\theta \wedge \beta$ satisfies
$$(\theta \wedge \beta) (x_0,\cdots,x_p) = \sum_{i=0}^{p}(-1)^i \theta(x_i)\beta(x_0,\cdots,\hat{x}_i,\cdots,x_p).$$
If $\beta = \varphi_1 \wedge \cdots \wedge \varphi_p$, then
$$(\theta \wedge \beta) (x_0,\cdots,x_p) = \det \begin{pmatrix}
        \theta(x_0) & \cdots & \theta(x_p)\\
        \varphi_1(x_0) & \cdots & \varphi_1(x_p)\\
        \vdots & & \vdots\\
        \varphi_p(x_0) & \cdots & \varphi_p(x_p)
    \end{pmatrix}.$$
By expanding the determinant along the first row, we get the equality.

\begin{remark}
    If $V$ is a free $K$-module generated by a basis $(e_i)_{i=1}^{n}$ be its dual basis.
    Then $(e_{i_1} \wedge \cdots \wedge e_{i_p})_{1\le i_1 < \cdots < i_p \le n}$ is a basis of $\bigwedge^p (V)$,
    and $(e_{i_1}^\vee \wedge \cdots \wedge e_{i_p}^\vee)_{1\le i_1 < \cdots < i_p \le n}$ is a basis of $\bigwedge^p (V^\vee)$.
    $(\Phi(e_i^\vee \wedge \cdots \wedge e_{i_p}^\vee))_{1\le i_1 < \cdots < i_p \le n}$ is the dual basis of $(e_{i_1} \wedge \cdots \wedge e_{i_p})_{1\le i_1 < \cdots < i_p \le n}$.
    So $\Phi$ is an isomorphism.
\end{remark}    

\begin{definitionenv}
    Let $X_1,\cdots,X_p$ be tangent vector fields on $U$, and $\omega$ be a $p$-form on $U$.
    Then $\omega(X_1,\cdots,X_p)$ denotes
    $$\begin{array}{rrcl}
        \omega(X_1,\cdots,X_p) : & U & \longrightarrow & \RR\\
        & x & \longmapsto & \omega(x)(X_1(x),\cdots,X_p(x)).
    \end{array}$$
\end{definitionenv}

\begin{propositionenv}
    Let $m \in \NN_{\ge 1} \cup \{\infty\}$.
    Let $X$ be a tangent vector field on $U$ of class $C^m$.
    For any $p$-form $\omega$ on $U$ of class $C^m$, there exists a unique $p$-form $\mathcal{L}_{X}(\omega)$ of class $C^{m-1}$ on $U$ such that
    $$\mathcal{L}_{X}(\omega)(X_1,\cdots,X_p) = X\left(\omega(X_1,\cdots,X_p)\right) - \sum_{i=1}^{p} \omega(X_1,\cdots,[X,X_i],\cdots,X_p)$$
    for any tangent vector fields $X_1,\cdots,X_p$ of class $C^1$ on $U$.
\end{propositionenv}
\begin{proofenv}
    We check that the term on the right-hand side, viewed as a function on $(X_1,\cdots,X_p)$ is $C^1(U)$-multilinear.
    Let $f \in C^1(U)$.
    If we replace $X_1$ by $f X_1$ on the right-hand side, we get
    $$\begin{aligned}
        & X\left(\omega(X_1,X_2,\cdots,f X_j,\cdots,X_p)\right)\\
        = & X \left(f \omega(X_1,\cdots,X_j,\cdots,X_p)\right)\\
        = & X(f) \omega(X_1,\cdots,X_j,\cdots,X_p) + f X\left(\omega(X_1,\cdots,X_j,\cdots,X_p)\right)
    \end{aligned}$$
    and the sum becomes,
    $$\begin{aligned}
        & \omega(X_1,\cdots,[X,f X_j],\cdots,X_p) + \sum_{\substack{i \in \{1,\cdots,n\}\\i \neq j}} f \omega(X_1,\cdots,[X,X_i],\cdots,X_p)\\
        = & X(f) \omega(X_1,\cdots,X_p) + \sum_{i=1}^{n} \omega(X_1,\cdots,[X,X_i],\cdots,X_p).
    \end{aligned}$$
    Let $i$ and $j$ be indices in $\{1,\cdots,p\}$ such that $X_i = X_j$.
    Then $\omega(X_1,\cdots,X_p) = 0$ and
    $$\begin{aligned}
        & \sum_{k=1}^{n} \omega(X_1,\cdots,[X,X_k],\cdots,X_p)\\
        = & \omega(X_1,\cdots,[X,X_i],\cdots,X_p) + \omega(X_1,\cdots,[X,X_j],\cdots,X_p)\\
        = & 0.
    \end{aligned}$$
\end{proofenv}

\begin{notationenv}
    Let $E$ be a finite-dimensional vector space over $\RR$, $U \subseteq E$ be an open subset,
    $X$ be a tangent vector field on $U$ of class $C^m$.
    If $\omega$ is a $p$-form on $U$ of class $C^m$, 
    $$\mathcal{L}_{X}(\omega)(X_1,\cdots,X_p) = X\left(\omega(X_1,\cdots,X_p)\right) - \sum_{i=1}^{p} \omega(X_1,\cdots,[X,X_i],\cdots,X_p).$$
    $\mathcal{L}_{X}(\omega)$ is a $p$-form of class $C^{m-1}$.
\end{notationenv}

\begin{propositionenv}
    Let $X$ be a tangent vector field on $U$ of class $C^1$.
    Let $p$ and $q$ be natural numbers.
    $\alpha$ and $\beta$ be $p$-form and $q$-form of class $C^1$ on $U$.
    Then 
    $$\mathcal{L}_{X}(\alpha \wedge \beta) = \mathcal{L}_{X}(\alpha) \wedge \beta + \alpha \wedge \mathcal{L}_{X}(\beta).$$
\end{propositionenv}
\begin{proofenv}
    We reason by induction on $p$.
    For $p=0$, $\alpha$ is given by a mapping $f : U \rightarrow \RR$ of class $C^1$.
    By definition,
    $$\begin{aligned}
        & \mathcal{L}_{X}(f \beta)(X_1,\cdots,X_q)\\
        = & X(f\beta(X_1,\cdots,X_q)) - \sum_{i=1}^{q}f \beta(X_1,\cdots,[X,X_i],\cdots,X_q)\\
        = & f \mathcal{L}_{X}(\beta)(X_1,\cdots,X_q) + X(f) \beta(X_1,\cdots,X_q).
    \end{aligned}$$
    For $p = 1$,
    $$\begin{aligned}
        (\alpha \wedge \beta)(X_0,\cdots,X_q) = \sum_{i=0}^{q} (-1)^i \alpha(X_i) \beta(X_0,\cdots,\hat{X}_i,\cdots,X_q).
    \end{aligned}$$
    $$\begin{aligned}
        & \mathcal{L}_{X}(\alpha \wedge \beta) (X_0,\cdots,X_q)\\
        = & X \left((\alpha \wedge \beta)(X_0,\cdots,X_q)\right) - \sum_{i=0}^{q}(\alpha \wedge \beta)(X_0,\cdots,\hat{X}_i,\cdots,X_q)\\
        = & X\left(\sum_{i=0}^{q}(-1)^i \alpha (X_i) \beta(X_0,\cdots,\hat{X}_i,\cdots,X_q)\right)\\
        & - \sum_{i=0}^{q}((-1)^{i}\alpha([X,X_i]) \beta(X_0,\cdots,\hat{X}_i,\cdots,X_q)\\
        & \quad + \sum_{j \neq i} (-1)^j \alpha(X_j) \beta(X_0,\cdots,\hat{X}_j,\cdots,[X,X_i],\cdots,X_q))\\
        = & \sum_{i=0}^{q}(-1)^{i} \left(X(\alpha(X_i)) - \alpha([X,X_i])\right) \beta(X_0,\cdots,\hat{X}_i,\cdots,X_q)\\
        & + \sum_{j=0}^{q}(-1)^{j} \alpha(X_j) (X(\beta(X_0,\cdots,\hat{X}_j,\cdots,X_q))\\
        & \quad - \sum_{i \neq j} \beta(X_0,\cdots,\hat{X}_j,\cdots,[X,X_i],\cdots,X_q))\\
        = & (\mathcal{L}_{X}(\alpha) \wedge \beta)(X_0,\cdots,X_q) + (\alpha \wedge \mathcal{L}_{X}(\beta))(X_0,\cdots,X_q).
    \end{aligned}$$
    Assume that $p \ge 2$ and the formula holds for $p-1$-form and $q$-form.
    By multilinearity, we may assume that $\alpha = \theta \wedge \omega$, where $\theta$ is a $1$-form and $\omega$ is a $(p-1)$-form.
    $$\begin{aligned}
        & \mathcal{L}_{X}(\alpha \wedge \beta) = \mathcal{L}_{X}\left(\theta \wedge (\omega \wedge \beta)\right)\\
        = & \mathcal{L}_{X}(\theta)\wedge(\omega \wedge \beta) + \theta \wedge \mathcal{L}_{X}(\omega \wedge \beta)\\
        = & (\mathcal{L}_{X}(\theta)\wedge\omega) \wedge \beta + \theta \wedge (\mathcal{L}_{X}(\omega) \wedge \beta + \omega \wedge \mathcal{L}_{X}(\beta))\\
        = & (\mathcal{L}_{X}(\theta) \wedge \omega + \theta \wedge \mathcal{L}_{X}(\omega)) \wedge \beta + (\theta \wedge \omega)\wedge\mathcal{L}_X(\beta)\\
        = & \mathcal{L}_{X}(\theta \wedge \omega) \wedge \beta + (\theta \wedge \omega) \wedge \mathcal{L}_{X}(\beta).
    \end{aligned}$$
\end{proofenv}

\begin{definitionenv}
    Let $\omega$ be a $p$-form on $U$.
    For any tangent vector field $X$,
    we denote by $\iota_{X}(\omega)$ the $(p-1)$-form on $U$ such that
    $$\iota_{X}(\omega)(X_1,\cdots,X_{p-1}) = \omega(X,X_1,\cdots,X_{p-1}).$$
\end{definitionenv}

\begin{propositionenv}
    If $\alpha$ and $\beta$ are $p$-form and $q$-form on $U$, then
    $$\iota_{X}(\alpha \wedge \beta) = \iota_{X}(\alpha) \wedge \beta + (-1)^{p}\alpha \wedge \iota_{X}(\beta).$$
\end{propositionenv}
\begin{proofenv}
    We reason by induction on $p$.
    For $p=0$, By convention, $\iota_{X}(\alpha) = 0$.
    $\alpha$ is given by a mapping $f : U \rightarrow \RR$.
    $$\iota_{X}(f \beta)(X_1,\cdots,X_{q-1})
    = (f \beta)(X,X_1,\cdots,X_{q-1})
    = f \iota_{X}(\beta)(X_1,\cdots,X_{q-1}).$$
    For $p=1$, Let $X_0 = X$.
    $$\begin{aligned}
        & \iota_{X}(\alpha \wedge \beta)(X_1,\cdots,X_{q}) = (\alpha \wedge \beta)(X_0,X_1,\cdots,X_{q})\\
        = & \alpha(X_0) \beta(X_1,\cdots,X_{q}) + \sum_{i=1}^{q} (-1)^i \alpha(X_i) \beta(X_0,\cdots,\hat{X}_i,\cdots,X_{q})\\
        = & (\iota_{X}(\alpha) \wedge \beta)(X_1,\cdots,X_{q}) - (\alpha \wedge \iota_{X}(\beta))(X_1,\cdots,X_{q}).
    \end{aligned}$$
    Assume that $p\ge 2$ and the formula holds for $p-1$-form and $q$-form.
    Let $\alpha = \theta \wedge \omega$, where $\theta$ is a $1$-form and $\omega$ is a $(p-1)$-form.
    $$\begin{aligned}
        & \iota_{X}(\alpha \wedge \beta) = \iota_{X}\left(\theta \wedge (\omega \wedge \beta)\right)\\
        = & \iota_{X}(\theta) \wedge (\omega \wedge \beta) - \theta \wedge \iota_{X}(\omega \wedge \beta)\\
        = & (\iota_{X}(\theta) \wedge \omega) \wedge \beta - \theta \wedge (\iota_{X}(\omega) \wedge \beta + (-1)^{p - 1} \omega \wedge \iota_{X}(\beta))\\
        = & \left(\iota_{X}(\theta) \wedge \omega - \theta \wedge \iota_{X}(\omega)\right) \wedge \beta + (-1)^{p}\theta \wedge \omega \wedge \iota_{X}(\beta)\\
        = & \iota_{X}(\theta \wedge \omega) \wedge \beta + (-1)^{p} (\theta \wedge \omega) \wedge \iota_{X}(\beta).
    \end{aligned}$$
\end{proofenv}

\begin{theoremenv}[Cartan's magic formula]
    Let $E$ be a finite-dimensional vector space over $\RR$, $U \subseteq E$ be an open subset,
    $X$ be a vector field on $U$ of class $C^1$.
    For any $p$-form of class $C^1$ on $U$, one has
    $$\mathcal{L}_{X}(\omega) = \iota_{X}(\dd \omega) + \dd (\iota_{X} (\omega)).$$
\end{theoremenv}
\begin{proofenv}
    $$\begin{aligned}
        & \delta_{X}(\alpha \wedge \beta) = \iota_{X}(\dd (\alpha \wedge \beta)) + \dd (\iota_{X}(\alpha \wedge \beta))\\
        = & \iota_{X}(\dd \alpha \wedge \beta + (-1)^{p} \alpha \wedge \dd \beta) + \dd \left(\iota_{X}(\alpha) \wedge \beta + (-1)^{p}\alpha \wedge \iota_{X}(\beta)\right)\\
        = & \iota_{X}(\dd \alpha) \wedge \beta + (-1)^{p+1} \dd \alpha \wedge \iota_{X}(\beta)\\
        & + (-1)^{p}\iota_{X}(\alpha) \wedge \dd \beta + \alpha \wedge \iota_{X}(\dd \beta)\\
        & + \dd (\iota_{X}(\alpha)) \wedge \beta + (-1)^{p-1} \iota_{X}(\alpha) \wedge \dd \beta\\
        & + (-1)^{p} \dd (\alpha) \wedge \iota_{X}(\beta) + \alpha \wedge \dd (\iota_{X}(\beta))\\
        = & \delta_{X}(\alpha) \wedge \beta + \alpha \wedge \delta_{X}(\beta).
    \end{aligned}$$
    If $\omega$ is a $1$-form, 
    $$\mathcal{L}_{X}(\omega)(Y) = X(\omega(Y)) - \omega([X,Y]),$$
    $$\begin{aligned}
        \delta_{X}(\omega)(Y)
        = & \iota_{X} (\dd \omega) (Y) + \dd (\iota_{X}(\omega))(Y)\\
        = & (\dd \omega)(X,Y) + \dd(\omega(X))(Y)\\
        = & \dd \omega (X,Y) - Y (\omega(X)).
    \end{aligned}$$
    To prove that $\mathcal{L}_{X}(\omega)(Y) = \delta_{X}(\omega)(Y)$, we need to check
    $$\dd \omega (X,Y) = X (\omega(Y)) - Y (\omega(X)) - \omega([X,Y]).$$
    For this purpose, we may assume that $\omega = f \dd g$, where $f \in C^{1}(U)$, $g \in E^*$.
    $$\begin{aligned}
        \dd \omega (X,Y) = & (\dd f \wedge \dd g)(X,Y)\\
        = & \dd f (X) \dd g (Y) - \dd f (Y) \dd g (X)\\
        = & X(f) Y(g) - Y(f) X(g)\\
    \end{aligned}$$
    $$\begin{aligned}
        & X(\omega(Y)) - Y(\omega(X)) - \omega([X,Y])\\
        = & X(f Y(g)) - Y(f X(g)) - f [X,Y](g)\\
        = & X(f) Y(g) + f X(Y(g)) - Y(f) X(g) - f Y(X(g)) - f [X,Y](g).
    \end{aligned}$$
\end{proofenv}

\begin{theoremenv}
    Let $E$ be a finite-dimensional vector space over $\RR$, $U \subseteq E$ be an open subset,
    $p \in \NN$, $\omega$ be a $p$-form on $U$ of class $C^1$.
    For any tangent vector fields $X_1,\cdots,X_p$ of class $C^1$ on $U$, one has
    $$\begin{aligned}
        \dd \omega (X_0,\cdots,X_p) = & \sum_{i = 0}^{p} (-1)^{i} X_i (\omega(X_0,\cdots,\hat{X}_i,\cdots,X_p))\\
        & + \sum_{0 \le i < j \le p} (-1)^{i+j} \omega([X_i,X_j],X_0,\cdots,\hat{X}_i,\cdots,\hat{X}_j,\cdots,X_p).
    \end{aligned}$$
\end{theoremenv}
\begin{proofenv}
    We reason by induction on $p$.
    For $p=0$, $\omega$ is given by a mapping $f : U \rightarrow \RR$ of class $C^1$.
    $$\dd f (X_0) = X_0(f).$$
    For $p=1$, this case have been checked in the proof of Cartan's magic formula.
    Suppose that the equality holds for $p-1$-form.
    $$\begin{aligned}
        \dd \omega (X_0,\cdots,X_p) = & \iota_{X_0} (\dd \omega) (X_1,\cdots,X_p)\\
        = & \mathcal{L}_{X_0} (\omega) (X_1,\cdots,X_p) - \dd (\iota_{X_0}(\omega))(X_1,\cdots,X_p).
    \end{aligned}$$
    $$\begin{aligned}
        & \mathcal{L}_{X_0} (\omega) (X_1,\cdots,X_p)\\
        = & X_0 (\omega(X_1,\cdots,X_p)) - \sum_{i=1}^{p} \omega(X_1,\cdots,[X_0,X_i],\cdots,X_p)\\
        = & X_0 (\omega(X_1,\cdots,X_p)) + \sum_{i=1}^{p} (-1)^{i} \omega ([X_0,X_i],X_1,\cdots,\hat{X}_i,\cdots,X_p).
    \end{aligned}$$
    By the induction hypothesis,
    $$\begin{aligned}
        & \dd (\iota_{X_0}(\omega))(X_1,\cdots,X_p)\\
        = & \sum_{i=1}^{p} (-1)^{i-1} X_i (\omega(X_0,X_1,\cdots,\hat{X}_i,\cdots,X_p))\\
        & - \sum_{0 \le i < j \le p} (-1)^{i+j} \omega([X_i,X_j],X_0,\cdots,\hat{X}_i,\cdots,\hat{X}_j,\cdots,X_p).
    \end{aligned}$$
\end{proofenv}

\begin{theoremenv}
    Let $E,F,G$ be Banach spaces over $\RR$, $U \subseteq E$, $V \subseteq F$ be open subsets, $(x_0,y_0) \in U \times V$.
    Let $f : U \times V \rightarrow G$ be a mapping.
    Let $n \in \NN_{\ge 1} \cup \{\infty\}$.
    Assume that
    \begin{enumerate}[label=(\arabic*)]
        \item $f(x_0,y_0) = 0$.
        \item $f$ is of class $C^n$ on a neighborhood of $(x_0,y_0)$.
        \item $\partial_{y} f(x_0,y_0) : F \rightarrow G$ is an isomorphism of Banach spaces.
    \end{enumerate}
    Then there exists an open subsets $U_0$ and $V_0$ of $E$ and $F$ such that $x_0 \in U_0 \subseteq U$, $y_0 \in V_0 \subseteq V$,
    and a unique mapping $\varphi : U_0 \rightarrow V_0$ of class $C^n$, such that $\varphi(x_0) = y_0$ and
    $$\forall x \in U_0, \quad f(x, \varphi(x)) = 0.$$
    Moreover, 
    $$\DD \varphi(x) = - \left(\partial_{y} f(x,\varphi(x))\right)^{-1} \circ \partial_{x} f(x,\varphi(x)).$$
\end{theoremenv}
\begin{proofenv}
    Let
    $$\begin{array}{rrcl}
        g : & U \times V & \longrightarrow & E \times G\\
            & (x,y) & \longmapsto & (x, f(x,y)).
    \end{array}$$
    $$\DD g (x_0,y_0)(\mu,\nu) = (\mu,\partial_{x} f(x_0,y_0)(\mu) + \partial_{y} f(x_0,y_0)(\nu)).$$
    This mapping is invertible, and its inverse is given by
    $$(a,b) \longmapsto (a,\partial_{y} f(x_0,y_0)^{-1}(b - \partial_{x} f(x_0,y_0)(a))).$$
    By the local inversion theorem, there exists an open neighborhood $W \subseteq E \times F$ of $(x_0,y_0)$, $W \subseteq U \times V$ such that $g|_{W}$ defines a $C^n$ homeomorphism.
    
    \quad Note that $g(x_0,y_0) = (x_0,0)$.
    There exists $U_0 \subseteq E$ open, $x_0 \in U_0 \subseteq U$ such that for any $x \in U_0$, $(x,0) \in g(W)$.
    So for any $x \in U_0$, there exists a unique $\varphi(x) \in V$ such that $(x,\varphi(x)) \in W$ and $f(x,\varphi(x)) = 0$.
\end{proofenv}

\begin{exampleenv}
    Consider
    $$\begin{array}{rrcl}
        f : & \RR^2 & \longrightarrow & \RR\\
        & (x,y) & \longmapsto & x^2 + y^2 - 1
    \end{array}$$
    and $(x_0,y_0) = (0,1)$, one easily checks
    $$\frac{\partial f}{\partial y} (x,y) = 2y, \quad \frac{\partial f}{\partial y} (x_0,y_0) = 2.$$
    Hence
    $$\varphi'(0) = \frac{\frac{\partial f}{\partial x}(0,1)}{\frac{\partial f}{\partial y}(0,1)} = 0.$$
\end{exampleenv}

            $$g : U \longrightarrow G$$
            to be the composite of $f$ with $\pi : E \longrightarrow G$.
            For any $x \in U$, $\DD f(x)$ is an isomorphism.
            Hence $\DD g(x) = \pi \circ \DD f (x)$ is surjective.
            We deduce that $g$ is a submersion.

            \quad One can easily check that
            $$g^{-1}(\{0\}) = f^{-1}\left(\pi^{-1}(\{0\})\right) = f^{-1} \left(F \cap V\right) = U \cap M.$$
        \item (2) $\Rightarrow$ (3).

            By definition, $\DD g(a) : E \longrightarrow G$ is surjective.
            Let $$H \coloneq \ker \DD g(a),$$ and let $u : G \longrightarrow E$ $\RR$ linear such that $\DD g(a) \circ U = \Id(G)$ (in particular $u$ is injective).
            
            \quad Note that $E = H \oplus u(G)$.
            Consider
            $$\begin{array}{rrcl}
                h : & H \times u(G) & \longrightarrow & G\\
                & (x,y) & \longmapsto & g (x + y + a)
            \end{array}$$
            and notice that
            $$\DD g(a) |_{u(G)} = \DD_y h(0,0)$$
            is an isomorphism.
            By implicit function theorem, there exists an open neighborhood $W$ of $0$ in $H$ and a mapping $\psi : W \longrightarrow u(G)$ of class $C^l$ such that
            $\psi(0) = 0$ and for any $x \in W$,
            $$h(x,\psi(x)) = g(x + \psi(x) + a) = 0.$$
            Namely,
            $$x + \psi(x) + a \in U \cap M.$$
            Let
            $$\begin{array}{rrcl}
                \varphi : & W & \longrightarrow & E\\
                & x & \longmapsto & x + \psi(x) + a.
            \end{array}$$
            Then $\varphi(0) = a$, $\DD \varphi(x) = i_{H} + \DD_{\psi}(x)$,
            where $i_H : H \longrightarrow E$ is the mapping of inclusion.
            Since $\DD \psi (x)$ takes values in $u(G)$, and $E = H \oplus u(G)$, so $\DD \varphi (x)$ is injective.
            Using $E = H \oplus u(G)$, we obtain that $\varphi$ is injective.
            We conclude that $\varphi$ is an immersion.
            Its inverse is the restriction of the projection to $H$ along $u(G)$ to the image of $\varphi$ which is also continuous.
            Hence $\varphi$ is a homeomorphism to its image.
        \item (3) $\Rightarrow$ (1).

            Without loss of generality, we may assume that $a = 0$.
            Let $F = \mathrm{Im}(\DD \varphi (0))$, $\dim (F) = d$.
            We choose a vector subspace $F'$ of $E$ such that $E = F \oplus F'$.
            Let $\pi : E \longrightarrow F$, $\pi' : E \longrightarrow F'$ be linear projections (in other words, if $x = u + u'$, $u \in F$, $u' \in F'$),
            then $\pi(x) = u$ and $\pi'(x) = u'$.
            By local inversion theorem, there exists an open subset $0 \in W_0 \subseteq W$ such that 
            $\pi\circ \varphi |_{W_0}$ is $C^l$-homomorphism.
            By ``shrinking'' $W$ (to $W_0$), we may assume $\pi \circ \varphi$ is $C^l$-homeomorphism.
            Take $f : U \longrightarrow E$, $f(x) = x - (\pi' \circ \varphi)((\pi \circ \varphi)^{-1} (\pi(x)))$, $V = \mathrm{Im}(f)$.
            Notice that
            $$f^{-1} : V \ni y \mapsto y + (\pi' \circ \varphi)\left((\pi \circ \varphi)^{-1} (\pi(y))\right)$$
            and both $f,f^{-1}$ are of class $C^l$.
            Hence $f$ is $C^l$-diffeomorphism.
            Finally, $x \in U \cap M$ if and only if there exists $b \in W$ such that $\varphi(b) = x$.
            In this case,
            $$f(x) = x - (\pi' \circ \varphi)(b) = x - \pi'(x) = \pi(x) \in V \cap F.$$
            Conversely, if $f(x) \in V \cap F$, then
            $$(\pi' \circ \varphi)\left((\pi \circ \varphi)^{-1} (\pi(x))\right) = \pi'(x).$$
            Since $\pi(\varphi(\pi \circ \varphi)^{-1} (\pi (x))) = \pi(x)$,
            we have
            $$x = \pi'(x) + \pi(x) = \varphi\left((\pi \circ \varphi)^{-1}(\pi(x))\right).$$
    \end{enumerate}
\end{proofenv}

\begin{definitionenv}
    Let $E$ be a finite dimensional vector space over $\RR$, and $M$ be a subset of $E$.
    Let $l \in \NN_{\ge 1} \cup \{\infty\}$.
    If $M$ satisfies one of the $3$ conditions of the previous theorem,
    then we say that $M$ is a $C^l$ submanifold of dimension $d$ of $E$.
\end{definitionenv}

\begin{exampleenv}
    Let $E$ be a finite dimensional vector space over $\RR$, $U \subseteq E$ open, $U$ is a $C^\infty$ submanifold of $E$ of dimension $n$.
\end{exampleenv}

\begin{exampleenv}
    Let $E$ be a finite dimensional vector space over $\RR$, $U \subseteq E$ open, $f : U \longrightarrow \RR$ a mapping,
    $M \coloneq \{x \in U \mid f(x) = 0\}$ and assume that $\DD f(x)$ is non-zero for any $x \in U$. Then $M$ is a submanifold of $E$ called a hypersurface in $E$ ($a_1 x_1 + \cdots + a_n x_n = b$ hyperplane).
\end{exampleenv}

\begin{theoremenv}
    Let $E$ be a finite-dimensional vector space over $\RR$, let $l \in \NN_{\ge 1} \cup \{\infty\}$ and $M$ be a submanifold of class $C^l$ of $E$ of dimension $d$.
    Let $a \in M$.
    Then the tangent space $T_aM$ is a vector subspace of $E$ of dimension $d$.
    Moreover,
    \begin{enumerate}[label=(\arabic*)]
        \item Suppose that $U$ is an open neighborhood of $a$, $V$ is an open neighborhood of $0$, $F$ is a vector subspace of $E$ of dimension $d$, and $f : U \longrightarrow V$ is a $C^l$ diffeomorphism such that $f(a) = 0$ and $f(U \cap M) = F \cap V$.
            Then
            $$T_aM = \left(\DD f(a)\right)^{-1}(F).$$
        \item Suppose that $U$ is an open neighborhood of $a$, $G$ is a vector space of dimension $n - d$ over $\RR$, and $g : U \longrightarrow G$ is a submersion of class $C^l$ such that
            $$U \cap M = g^{-1}(\{0\}),$$
            then
            $$T_aM = \ker\left(\DD g(a)\right).$$
        \item Suppose that $U$ is an open neighborhood of $a$ in $E$,
            $H$ is a vector space of dimension $d$ over $\RR$, $W \subseteq H$ open, $\varphi : W \longrightarrow E$ is an immersion of class $C^l$ such that
            $a \in \varphi(W)$ and such that $\varphi$ defines a homeomorphism between $W$ and $U \cap M$.
            Then one has
            $$T_a M = \DD \varphi\left(\varphi^{-1}(a)\right)(H).$$
    \end{enumerate}
